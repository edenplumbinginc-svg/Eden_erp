Step: Create a /showcase page that auto-lists every required route from docs/ui-contract.yaml (no hardcoding) and links to them.

Why it matters: A single visual “see-it-all” surface lets you click everything you’ve built and instantly spot gaps or broken links.

Inputs needed:

Existing docs/ui-contract.yaml

React/Vite app with React Router (paths below assume src/)

Command (copy–paste):

# 1) Generator: YAML → JSON for the browser
mkdir -p scripts src/showcase
cat > scripts/generate-showcase.js << 'JS'
const fs = require('fs');
const path = require('path');
const yaml = require('js-yaml');

const specPath = path.join(process.cwd(), 'docs/ui-contract.yaml');
const outDir = path.join(process.cwd(), 'src', 'showcase');
const outFile = path.join(outDir, 'routes.json');

const doc = yaml.load(fs.readFileSync(specPath, 'utf8'));
const set = new Set();
for (const res of doc.resources || []) {
  for (const r of res.required_pages || []) {
    // inflate /path/[id] → /path/123 for demo navigation
    set.add(String(r).replace(/\[.*?\]/g, '123'));
  }
}

const routes = [...set].sort().map(route => ({
  route,
  label: route === '/' ? 'Home' : route
}));

fs.mkdirSync(outDir, { recursive: true });
fs.writeFileSync(outFile, JSON.stringify({ routes }, null, 2), 'utf8');
console.log(`Wrote ${outFile} with ${routes.length} routes.`);
JS

# 2) Page component
cat > src/showcase/Showcase.jsx << 'JSX'
import { useEffect, useState } from 'react';
import { Link } from 'react-router-dom';

export default function Showcase() {
  const [routes, setRoutes] = useState([]);

  useEffect(() => {
    // routes.json is generated at build/dev start
    import('./routes.json').then(m => setRoutes(m.routes || [])).catch(() => setRoutes([]));
  }, []);

  if (!routes.length) {
    return <div data-state="loading" className="skeleton">Loading showcase…</div>;
  }

  return (
    <div style={{ padding: '1rem' }}>
      <h1>Showcase</h1>
      <p>Click any route to smoke it visually. Parametrized routes are inflated to demo values.</p>
      <div style={{
        display: 'grid',
        gridTemplateColumns: 'repeat(auto-fill, minmax(220px, 1fr))',
        gap: '12px'
      }}>
        {routes.map(({ route, label }) => (
          <Link key={route} to={route} style={{
            display:'block',
            padding:'14px 16px',
            borderRadius:'12px',
            border:'1px solid #e5e7eb',
            textDecoration:'none'
          }}>
            <div style={{ fontSize:'0.8rem', opacity:0.7 }}>Route</div>
            <div style={{ fontWeight:600 }}>{label}</div>
          </Link>
        ))}
      </div>
    </div>
  );
}
